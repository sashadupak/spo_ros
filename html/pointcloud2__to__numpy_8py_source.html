32 def pointcloud2_to_dtype(cloud_msg):
33  '''!
34  Конвертирует список PointField в список элементов numpy.uint8
35  @param cloud_msg: список типа PointField
36  @return: возвращает список типа numpy.uint8
37  '''
38  offset = 0
39  np_dtype_list = []
40  for f in cloud_msg.fields:
41  while offset < f.offset:
42  # может быть доп. отступ между полями
43  np_dtype_list.append(('%s%d' % (PREFIX, offset), np.uint8))
44  offset += 1
45  np_dtype_list.append((f.name, pftype_to_nptype[f.datatype]))
46  offset += pftype_sizes[f.datatype]
47 
48  # может быть доп. отступ между полями
49  while offset < cloud_msg.point_step:
50  np_dtype_list.append(('%s%d' % (PREFIX, offset), np.uint8))
51  offset += 1
52 
53  return np_dtype_list

74 def pointcloud2_to_array(cloud_msg, split_rgb=False, remove_padding=True):
75  '''!
76  Конвертирует ros2 PointCloud2 сообщение в список элементов numpy.uint8
77 
78  Изменяет форму возвращаемого массива (высота, ширина), даже если первоначальная высота равна 1
79 
80  @param cloud_msg: ros2 sensor_msgs.msg.PointCloud2
81  @param split_rgb: если True, то вызывает метод split_rgb_field
82  @param remove_padding: если True, то убирает отступы
83  @return: возвращает массив (список) элементов numpy.uint8
84  '''
85  # construct a numpy record type equivalent to the point type of this cloud
86  dtype_list = pointcloud2_to_dtype(cloud_msg)
87 
88  # parse the cloud into an array
89  cloud_arr = np.fromstring(bytes(cloud_msg.data), dtype_list)
90 
91  # remove the dummy fields that were added
92  if remove_padding:
93  cloud_arr = cloud_arr[
94  [fname for fname, _type in dtype_list if not (fname[:len(PREFIX)] == PREFIX)]]
95 
96  if split_rgb:
97  cloud_arr = split_rgb_field(cloud_arr)
98 
99  return np.reshape(cloud_arr, (cloud_msg.height, cloud_msg.width))

102 def split_rgb_field(cloud_arr):
103  '''!
104  Разбивает массив с полем rgb типа float32 на массив с тремя полями r,g,b каждое из которых uint8
105  @param cloud_arr: массив float32 с полем rgb
106  @returns: массив uint8 c полями r,g,b
107  '''
108  rgb_arr = cloud_arr['rgb'].copy()
109  rgb_arr.dtype = np.uint32
110  r = np.asarray((rgb_arr >> 16) & 255, dtype=np.uint8)
111  g = np.asarray((rgb_arr >> 8) & 255, dtype=np.uint8)
112  b = np.asarray(rgb_arr & 255, dtype=np.uint8)
113 
114  # create a new array, without rgb, but with r, g, and b fields
115  new_dtype = []
116  for field_name in cloud_arr.dtype.names:
117  field_type, field_offset = cloud_arr.dtype.fields[field_name]
118  if not field_name == 'rgb':
119  new_dtype.append((field_name, field_type))
120  new_dtype.append(('r', np.uint8))
121  new_dtype.append(('g', np.uint8))
122  new_dtype.append(('b', np.uint8))
123  new_cloud_arr = np.zeros(cloud_arr.shape, new_dtype)
124 
125  # fill in the new array
126  for field_name in new_cloud_arr.dtype.names:
127  if field_name == 'r':
128  new_cloud_arr[field_name] = r
129  elif field_name == 'g':
130  new_cloud_arr[field_name] = g
131  elif field_name == 'b':
132  new_cloud_arr[field_name] = b
133  else:
134  new_cloud_arr[field_name] = cloud_arr[field_name]
135  return new_cloud_arr

138 def get_xyz_points(cloud_array, remove_nans=True, dtype=np.float):
139  '''!
140  Преобразует массив с полями x,y,z в матрицу 3хN
141  @param cloud_array: список элементов numpy
142  @param dtype: тип элементов
143  @returns: матрица 3хN
144  '''
145  # remove crap points
146  if remove_nans:
147  mask = np.isfinite(cloud_array['x']) & np.isfinite(cloud_array['y']) & np.isfinite(cloud_array['z'])
148  cloud_array = cloud_array[mask]
149 
150  # pull out x, y, and z values
151  points = np.zeros(list(cloud_array.shape) + [3], dtype=dtype)
152  points[...,0] = cloud_array['x']
153  points[...,1] = cloud_array['y']
154  points[...,2] = cloud_array['z']
155 
156  return points
